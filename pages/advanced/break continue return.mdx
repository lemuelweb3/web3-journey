## Different Types Of Javascript Functions

```js filename="functions.js" {3} copy
// -------- 

console.log("For loop with a break")
 for (let i=0; i < 11; i++) {
  if (i === 6) {
    break // Exit the loop when i is equal to 6
  }
  console.log(i) // output is 0, 1, 2, 3, 4, 5
 }

// -------- 
 
console.log("For loop with continue")
 for (let i=0; i < 11; i++) {
  if (i % 2 == 0) {
    continue // skip if i is an even number (divisible by 2)
  }
  console.log(i) // output is 1, 3, 5, 7, 9
 }

// -------- 

console.log("return statement in javascript")
 function cube(x) {
  return x*x*x
 }

let result = cube(4) // result is equal to 64
console.log("cube: result: ", result)
console.log("")

```
