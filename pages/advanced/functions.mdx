# Advanced Javascript

## Different Types Of Javascript Functions

```js filename="functions.js" {3} copy
// --------
 
 
console.log('Here Is A Simple Function');
function greetOne() {
  console.log('sayHello: Hello!')
}
 
greetOne() // Output: Hello!
console.log('')
 
// --------
 
 
console.log('A Simple function with arguments')
function greetTwo(name) {
  console.log('sayHello: Hello ' + name + '!')
}
 
greetTwo()
console.log('')
 
// --------
 
console.log('A Function with return values')
function square(x) {
  return x * x
}
let result = square(7) // result is now equal to 49
console.log('square: result: ', result)
console.log('')
 
// --------
 
console.log('A Function with multiple arguments and return values')
function sumAndProduct(num1, num2, num3) {
  return [num1 + num2 + num3, num1 * num2 * num3]
}
 
// array destructuring
let [sum, product] = sumAndProduct(4, 5, 6)
console.log('sumAndProduct: sum: ', sum) // Output: 15
console.log('sumAndProduct: product: ', product) // Output 120
console.log('')
 
// --------
 



```
